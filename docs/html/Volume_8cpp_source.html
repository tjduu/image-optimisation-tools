20// custom comparison function for natural sorting
21bool naturalSort(const std::string& a, const std::string& b) {
22 std::regex re("(\\d+)");
23 std::sregex_iterator it_a(a.begin(), a.end(), re), it_b(b.begin(), b.end(), re);
24 std::sregex_iterator end;
25
26 // compare each matched number
27 while (it_a != end && it_b != end) {
28 int num_a = std::stoi(it_a->str());
29 int num_b = std::stoi(it_b->str());
30
31 if (num_a != num_b) {
32 return num_a < num_b;
33 }
34 ++it_a;
35 ++it_b;
36 }
37
38 // if all numbers are equal, sort based on original string
39 return a < b;
40}

42// custom sorting function for natural sorting
43void customSort(std::vector<std::string>& files) {
44 for (size_t i = 1; i < files.size(); ++i) {
45 std::string key = files[i];
46 int j = i - 1;
47
48 // 将files[i]移动到它在前面子数组中的正确位置
49 while (j >= 0 && naturalSort(key, files[j])) {
50 files[j + 1] = files[j];
51 j = j - 1;
52 }
53 files[j + 1] = key;
54 }
55}

150// Generate 3D volume data from slices
151void Volume::generate3DVolumeFromSlices() {
152 if (slices.empty()) {
153 throw std::runtime_error("Volume is empty, no slices to generate 3D data from.");
154 }
155
156 width = slices[0]->getWidth();
157 height = slices[0]->getHeight();
158 channels = slices[0]->getChannels();
159 depth = static_cast<int>(slices.size());
160
161 size_t singleImageSize = width * height * channels;
162 size_t totalSize = singleImageSize * depth;
163 data = new unsigned char[totalSize];
164 for (size_t i = 0; i < slices.size(); ++i) {
165 std::memcpy(data + i * singleImageSize, slices[i]->getData(), singleImageSize);
166 }
167}

189std::unique_ptr<Slice> Volume::getSliceXZ(int y) const {
190 if (y < 0 || y >= height) {
191 throw std::out_of_range("Y coordinate out of range.");
192 }
193
194 auto sliceData = std::make_unique<unsigned char[]>(width * depth * channels);
195 for (int z = 0; z < depth; ++z) {
196 for (int x = 0; x < width; ++x) {
197 for (int c = 0; c < channels; ++c) {
198 int volIndex = ((z * height + y) * width + x) * channels + c;
199 int sliceIndex = (z * width + x) * channels + c;
200 sliceData[sliceIndex] = data[volIndex];
201 }
202 }
203 }
204 return std::make_unique<Slice>(width, depth, channels, std::move(sliceData));
205}
